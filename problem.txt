// pode pedir para ir para o próximo estado

class State {
    currentMsg = null
    nextState = null
    constructor(Semaphore s){}

    getMsg(): return currentMsg

    changeState(){
        
    }
}

class RedState extends State {
    constructor(Semaphore s){
        nextState = GreenState
    }

    currentMsg = "ESPERE AÍ"
    changeState(){

    }

}

class GreenState extends State {
    constructor(Semaphore s){
        nextState = YellowState
    }

    currentMsg = "SIMBORA"
}

class YellowState extends State {
    constructor(Semaphore s){
        nextState = RedState
    }

    currentMsg = "FICA ESPERTO, QUERIDO"
}

class Semaphore {
    currentState: State

    constructor(){
        currentState = RedState
    }

    change(){
        currentState = currentState->nextState
    }


}
